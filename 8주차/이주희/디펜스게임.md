## 코드
```Java
import java.util.*;

class Solution {
    
    public int solution(int n, int k, int[] enemy) {
        if(k >= enemy.length) {     //무적권으로 충분
            return enemy.length;
        }
        
        Queue<Integer> pq = new PriorityQueue<>();	//오름차순 정렬
        for(int i=0; i<k; i++) {
            pq.add(enemy[i]);
        }
        
        int cnt = k;                    //무적권의 갯수만큼은 커버 가능
        for(int i=k; i<enemy.length; i++) {
            if(pq.peek() < enemy[i]) {  //현재 적이 더 크므로 무적권으로 처치, pq에 있던 적은 남은 병사로 처치
                n -= pq.poll();
                pq.add(enemy[i]);
            } else {                    // 현재 적을 병사로 처치
                n -= enemy[i];
            }
            
            if(n<0) return cnt;		//현재 적을 처치 못함
            if(n==0) return cnt+1;
            cnt+=1;     //병사로 처치한 적의 수 증가
        }
        
        return enemy.length;        //모든 적을 처치
    }
}
```

### 시도한 코드
-> 런타임 에러, 시간 초과    
```Java
class Solution {
	    int answer;
	    boolean res;
	    public int solution(int n, int k, int[] enemy) {
	        answer = 0;
	        res = false;
	        process(0, n, k, enemy);
	        return answer;
	    }
	    void process(int idx, int left, int k, int[] enemy) {
	        if(res) return;
	        if(idx == enemy.length) {
	            answer = Math.max(answer, idx);
	            res = true;
	            return;
	        }
	        
	        if(enemy[idx] <= left) {
	            process(idx+1, left-enemy[idx], k, enemy);
	            if(k > 0) {
	                process(idx+1, left, k-1, enemy);
	            }
	        } else {
	            if(k > 0) {
	                process(idx+1, left, k-1, enemy);
	            } else {
	                answer = Math.max(answer, idx);
	                return;
	            }
	        }
	    }
	}
```
