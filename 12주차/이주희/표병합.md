import java.util.*;

class Solution {
    int N;
    int[] parent;
    String[] board;
    
    void initParent() {
        int size = N*N;
        parent = new int[size+1];
        for(int i=1; i<=size; i++) {
            parent[i] = i;              //자기 자신으로 초기화
            board[i] = "";
        }
    }
    
    void union(int r1, int c1, int r2, int c2) {
        if(r1 == r2 && c1 == c2) return;        //같은 셀
        
        int p1 = getParent(getNum(r1, c1));
        int p2 = getParent(getNum(r2, c2));
        
        String value = board[p1] == "" ? board[p2] : board[p1];
        parent[p2] = p1;
        board[p1] = value;
        board[p2] = "";
    }
    
    int getParent(int x) {
        if(parent[x] == x) return x;
        return parent[x] = getParent(parent[x]);
    }
    
    int getNum(int r, int c) {
        return (r-1)*N+c;
    }
    
    public String[] solution(String[] commands) {
        List<String> answer = new ArrayList<>();
        N = 50;
        board = new String[N*N+1];
        initParent();
        
        for(String command : commands) {
            String[] word = command.split(" ");
            int r, c, p;
            String value;
            switch(word[0]) {
                case "UPDATE":
                    if(word.length == 4) {      //UPDATE r c value
                        r = Integer.parseInt(word[1]);
                        c = Integer.parseInt(word[2]);
                        p = getParent(getNum(r, c));
                        board[p] = word[3];
                    } else {                    //UPDATE value1 value2
                        for(int i=1, size=N*N; i<=size; i++) {
                            if(board[i].equals(word[2])) {
                                board[i] = word[2]; 
                            }
                        }
                    }
                    break;
                case "MERGE":
                    int r1 = Integer.parseInt(word[1]);
                    int r2 = Integer.parseInt(word[2]);
                    int c1 = Integer.parseInt(word[3]);
                    int c2 = Integer.parseInt(word[4]);
                    
                    union(r1, r2, c1, c2);
                    break;
                case "UNMERGE":
                    r = Integer.parseInt(word[1]);
                    c = Integer.parseInt(word[2]);
                    int num = getNum(r, c);
                    p = getParent(num);
                    value = board[p];      
                    board[p] = "";
                    for(int i=1, size=N*N; i<=size; i++) {
                        if(parent[i] == p) {
                            parent[i] = i;
                        }
                    }
                    board[num] = value;
                    break;
                case "PRINT":
                    r = Integer.parseInt(word[1]);
                    c = Integer.parseInt(word[2]);
                    p = getParent(getNum(r, c));
                    value = board[p];
                    if(value == "")
                        value = "EMPTY";
                    answer.add(value);
                    break;
            }            
        }
        
        return answer.toArray(new String[answer.size()]);
    }
}
