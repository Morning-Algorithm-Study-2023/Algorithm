## 코드
```Java
import java.util.*;

class Solution {
    public String solution(int n, int k, String[] cmd) {
        List<Integer> ll = new LinkedList<Integer>();   //남은 행번호 리스트   
        char answer[] = new char[n];
        for(int i=0; i<n; i++) {
            ll.add(i);     
            answer[i] = 'O';
        }
        Stack<Integer> removed = new Stack<Integer>();  //제거된 행번호
        
        int nowIdx = k;
        for(String c : cmd) {
            String[] dir = c.split(" ");
            
            switch(dir[0]) {
                case "D":
                    nowIdx += Integer.parseInt(dir[1]);
                    break;
                case "U":
                    nowIdx -= Integer.parseInt(dir[1]);
                    break;
                case "C":
                    removed.add(ll.get(nowIdx));
                    answer[ll.get(nowIdx)] = 'X';
                    ll.remove(nowIdx);
                    if(nowIdx >= ll.size()) {
                        nowIdx = ll.size()-1;
                    }
                    break;
                case "Z":
                    int number = removed.pop();
                    answer[number] = 'O';
                    int idx = getIdx(number, ll);
                    ll.add(idx, number);
                    if(idx <= nowIdx) {
                        nowIdx++;
                    }
                    break;
            }
        }
        
        return String.valueOf(answer);
    }
    
    int getIdx(int target, List<Integer> ll) {
        int start = 0, end = ll.size()-1;
        
        int answer = 0;
        while(start <= end) {
            int mid = (start+end)/2;
            
            if(ll.get(mid) < target) {
                start = mid+1;
            } else {
            	answer = mid;
                end = mid-1;
            }
        }
        
        if(start >= ll.size()) answer = ll.size();
        return answer;
    }
    
}
```
