### 풀이
1. noTarget 함수를 통해 현 위치에서 가장 가까운 카드 선택
2. targrt 함수를 통해 선택된 카드까지 최단거리 구하기

```java
import java.util.*;

class Solution {
    int[] dx={-1,1,0,0};
    int[] dy={0,0,-1,1};
    int curR;
    int curC;
    int answer=0;
    
    int[][] graph;
    
    public int noTarget(int r, int c){
        Queue<Point> queue=new LinkedList<>();
        int[][] visited=new int[4][4];
        queue.add(new Point(r,c));
        while(!queue.isEmpty()){
            Point cur=queue.poll();
            
            // return 타켓값
            if(graph[cur.x][cur.y]!=0){
                int temp = graph[cur.x][cur.y];
                graph[cur.x][cur.y]=0;
                curR=cur.x;
                curC=cur.y;
                answer+=visited[cur.x][cur.y];
                answer++;
                return temp;
            }
            
            for(int i=0; i<4; i++){
                int nx=cur.x+dx[i];
                int ny=cur.y+dy[i];
                if(nx>=0 && nx<4 && ny>=0 && ny<4){
                    if(visited[nx][ny]==0){
                        visited[nx][ny]=visited[cur.x][cur.y]+1;
                        queue.add(new Point(nx,ny));
                    }
                }
            }
            
            // ctrl 연산
            // 있으면 멈추고 없으면 끝값
            for(int i=0; i<4; i++){
                int nx=cur.x;
                int ny=cur.y;
                while(true){
                    nx+=dx[i];
                    ny+=dy[i];
                    if(nx>=0 && nx<4 && ny>=0 && ny<4){
                        if(graph[nx][ny]!=0){
                            break;
                        }
                    }
                    else{
                        nx-=dx[i];
                        ny-=dy[i];
                        break;
                    }
                }
                if(nx!=cur.x || ny!=cur.y){
                    if(visited[nx][ny]==0){
                        visited[nx][ny]=visited[cur.x][cur.y]+1;
                        queue.add(new Point(nx,ny));
                    }
                }
            }
            
        }
        return 0;
    }
    
    public int target(int targetNum, int r, int c){
        Queue<Point> queue=new LinkedList<>();
        int[][] visited=new int[4][4];
        queue.add(new Point(r,c));
        while(!queue.isEmpty()){
            Point cur=queue.poll();
            
            // return 타켓값
            if(graph[cur.x][cur.y]==targetNum){
                int temp = graph[cur.x][cur.y];
                graph[cur.x][cur.y]=0;
                curR=cur.x;
                curC=cur.y;
                answer+=visited[cur.x][cur.y];
                answer++;
                return temp;
            }
            
            for(int i=0; i<4; i++){
                int nx=cur.x+dx[i];
                int ny=cur.y+dy[i];
                if(nx>=0 && nx<4 && ny>=0 && ny<4){
                    if(visited[nx][ny]==0){
                        visited[nx][ny]=visited[cur.x][cur.y]+1;
                        queue.add(new Point(nx,ny));
                    }
                }
            }
            
            // ctrl 연산
            // 있으면 멈추고 없으면 끝값
            for(int i=0; i<4; i++){
                int nx=cur.x;
                int ny=cur.y;
                while(true){
                    nx+=dx[i];
                    ny+=dy[i];
                    if(nx>=0 && nx<4 && ny>=0 && ny<4){
                        if(graph[nx][ny]!=0){
                            break;
                        }
                    }
                    else{
                        nx-=dx[i];
                        ny-=dy[i];
                        break;
                    }
                }
                if(nx!=cur.x || ny!=cur.y){
                    if(visited[nx][ny]==0){
                        visited[nx][ny]=visited[cur.x][cur.y]+1;
                        queue.add(new Point(nx,ny));
                    }
                }
            }
        }
        return 0;
    }
    
    
    public int solution(int[][] board, int r, int c) {
        graph=new int[4][4];
        curR=r;
        curC=c;
        
        int left=0;
        for(int i=0; i<4; i++){
            for(int j=0; j<4; j++){
                graph[i][j]=board[i][j];
                if(board[i][j]!=0) left++;
            }
        }
        
        left/=2;
        while(left-- > 0){
            int targetNum = noTarget(curR, curC);
            target(targetNum, curR, curC);
        }
        
        
        return answer;
    }
    
    class Point{
        int x;
        int y;
        
        Point(int x, int y){
            this.x=x;
            this.y=y;
        }
    }
}
```
