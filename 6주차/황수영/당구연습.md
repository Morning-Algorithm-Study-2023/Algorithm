import java.util.*;

class Solution {
    
    public int getHalfResult(int x, int y) {
        int tmp = (x*x) + (y*y);
        // tmp *= 2;
        // tmp *= 2;
        return tmp;
    }
    
    public int[] solution(int m, int n, int startX, int startY, int[][] balls) {
        int[] answer = new int[balls.length];
                
        for (int i = 0; i < balls.length; i++) {
            int[] tmp = new int[4];
            int x = balls[i][0];
            int y = balls[i][1];
            
            // 왼쪽
            int diffX = x;
            int diffY = Math.abs(y - startY)/2;
            int diffStartX = startX;
            int diffStartY = diffY;
            
            int first = getHalfResult(diffX, diffY);
            int second = getHalfResult(diffStartX, diffStartY);
            tmp[0] = (first + second) * 2;
            
            // 아래
            diffX = Math.abs(x - startX)/2;
            diffY = y;
            diffStartX = diffX;
            diffStartY = startY;
            
            first = getHalfResult(diffX, diffY);
            second = getHalfResult(diffStartX, diffStartY);
            tmp[1] = (first + second) * 2;
                
            // 위
            diffX = n - x;
            diffY = Math.abs(y - startY)/2;
            diffStartX = n - startX;
            diffStartY = diffY;
            
            first = getHalfResult(diffX, diffY);
            second = getHalfResult(diffStartX, diffStartY);
            tmp[2] = (first + second) * 2;
            
            // 오른쪽
            diffX = Math.abs(x - startX)/2;
            diffY = m - y;
            diffStartX = diffX;
            diffStartY = m - startY;
            
            first = getHalfResult(diffX, diffY);
            second = getHalfResult(diffStartX, diffStartY);
            tmp[3] = (first + second) * 2;
            
            // 그중에 제일 작은 값으로
            int tmpMin = Integer.MAX_VALUE;
            for (int j = 0; j < 4; j++) {
                tmpMin = Math.min(tmpMin, tmp[i]);
            }
            answer[i] = tmpMin;
        }
        
        return answer;
    }
}
