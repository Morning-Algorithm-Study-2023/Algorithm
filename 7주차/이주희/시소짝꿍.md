## 코드1
```Java
import java.util.*;

class Solution {
    public long solution(int[] weights) {
        Arrays.sort(weights);
        int N = weights.length;
        long answer = 0;
        
        int cnt = 0;
        for(int i=0; i<N; i++) {
            if(i > 0 && weights[i] == weights[i-1]) {
                cnt-=1;
                answer+=cnt;
                continue;
            }
            cnt=0;
            for(int j=i+1; j<N; j++) {
                if(isOk(weights[i], weights[j])) {
                    cnt++;
                }
            }
            answer+=cnt;
        }
        
        return answer;
      
    }
    
    boolean isOk(int num1, int num2) {
        if(num1==num2 || num1*4==num2*3 || num1*4==num2*2 || num1*3==num2*2) return true;
        return false;
    }
}
```

## 코드2
```Java
import java.util.*;

class Solution {
    public long solution(int[] weights) {
        Arrays.sort(weights);
        int N = weights.length;
        long answer = 0;
        
        int cnt = 0;
        for(int i=0; i<N; i++) {
            if(i > 0 && weights[i] == weights[i-1]) {
                cnt-=1;
                answer+=cnt;
                continue;
            }
            cnt=0;
            for(int j=i+1; j<N; j++) {
                int res = gcd(weights[i], weights[j]);
                int res1 = weights[i]/res, res2 = weights[j]/res;
                if((res1==1 && res2==1) || (res1==1 && res2==2) || (isOk(res1) && isOk(res2))) {
                    cnt++;
                }
            }
            answer+=cnt;
        }
        
        return answer;
      
    }
    
    boolean isOk(int num) {
        if(num==2 || num ==3 || num == 4)
            return true;
        return false;
    }
    
    int gcd(int a, int b) {
        if(b==0) return a;
        return gcd(b, a%b);
    }
}

```
