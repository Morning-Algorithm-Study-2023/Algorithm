import java.util.*;

class Solution {
    public int solution(int[][] jobs) {
        int answer = 0;
        // 1. 시작하는 순서대로 오름차순 정렬
        Arrays.sort(jobs, (o1, o2) -> o1[0] - o2[0]);
            
        // 2. 소요 시간순대로 오름차순 정렬
        PriorityQueue<int[]> pq = new PriorityQueue<>((o1, o2) -> o1[1] - o2[1]);
        
        int idx = 0;                // 수행할 작업의 인덱스(이미 다 시작순대로 정렬되어 있음)
        int endTime = jobs[0][0];   // 현재 작업이 끝나는 시간대
        
        // 3. 작업 -> 새로 큐에 들어갈 작업이 아직 남아있거나, 큐에 작업이 있거나
        while(idx < jobs.length || !pq.isEmpty()) {
            
            // 현재 작업 중, 요청이 온 애들은 다 넣기! -> 소요 시간 순서대로 정렬됨
            while (idx < jobs.length && jobs[idx][0] <= endTime) {
                pq.add(jobs[idx]);
                idx++;
            }
            
            // 이전 작업이 다 완료된 이후에나 새 요청이 들어옴 (큐에 새로 넣을 게 있는 지 확인)
            if (pq.isEmpty()) {
                endTime = jobs[idx][0];
                pq.add(jobs[idx++]);
            }
            
            // 이전 작업 수행 도중 들어온 요청이 있을 경우 (큐에 있을 경우)
            // 하나씩 작업 수행!
            int[] work = pq.poll();
            endTime += work[1];             // 끝나는 시간 계산하기
            answer += endTime - work[0];    // 현재 일의 {소요 시간 + 작업시간} 더해주기 => 전체 끝나는 시간 - 원래 시작 시간
        }
        
        return answer/jobs.length;
    }
}
