## 코드
```Java
import java.util.*;

class Solution {
    int[] dx = {-1, -1, 0, 1, 1, 1, 0, -1};
    int[] dy = {0, 1, 1, 1, 0, -1, -1, -1};
    
    class Node {
        int x, y;
    }

    public int solution(int[] arrows) {
        int answer = 0;
        int o1x=0, o1y=0;
        // 방문 여부 체크
        Map<String, ArrayList<String>> visited = new HashMap<>();
        visited.put(o1x+""+o1y, new ArrayList<>());
        
        for(int arrow : arrows) {
            for(int i=0; i<2; i++) {    // 대각선 체크를 위해 두 칸 이동
                int o2x = o1x + dx[arrow];
                int o2y = o1y + dy[arrow];
                
                String point1 = o1x+""+o1y;
                String point2 = o2x+""+o2y;
                ArrayList<String> list = visited.get(point2);
                if(list == null) {
                    list = new ArrayList<>();
                    list.add(point1);
                    visited.put(point2, list);
                } else {
                    boolean isOk = true;
                    for(String p : list) {
                        if(point2.equals(p)) {
                            isOk = false;
                            break;
                        }
                    }
                    if(isOk) {
                        answer++;
                        list.add(point2);
                        
                        list = visited.get(point2);
                        list.add(point1);
                    }
                    
                }
                o1x = o2x; o1y = o2y;
            }
        }
        
        return answer;
    }
}
```
