## 코드1
```Java
import java.util.*;

class Solution {
    public int solution(int[][] rectangle, int characterX, int characterY, int itemX, int itemY) {
        int[][] board = new int[101][101];
        for(int[] rec : rectangle) {
            fill(rec[0]*2, rec[1]*2, rec[2]*2, rec[3]*2, board);
        }
        
        int res = bfs(characterX, characterY, itemX, itemY, board);
        return res/2;
    }
    
    void fill(int x1, int y1, int x2, int y2, int[][] board) {
        for(int x=x1; x<=x2; x++) {
            for(int y=y1; y<=y2; y++) {
                if(board[x][y]==2) continue;    // 사각형 내부

                board[x][y] = 2;
                if(x==x1 || y==y2 || x==x2 || y==y1) {  // 테두리
                    board[x][y] = 1;
                }
            }
        }
    }
    
    int bfs(int x, int y, int tx, int ty, int[][] board) {
        int answer = Integer.MAX_VALUE;
        x*=2; y*=2; tx*=2; ty*=2;
        
        int[] dx = {-1, 1, 0, 0}; //↑, ↓, ←, →
        int[] dy = {0, 0, -1, 1};
        int[][] visited = new int[101][101];
        Queue<int[]> que = new ArrayDeque<>();
        que.add(new int[]{x, y});
        visited[x][y] = 1;
        
        while(!que.isEmpty()) {
            int[] now = que.poll();
            if(now[0]==tx && now[1]==ty) {
                answer = Math.min(answer, visited[now[0]][now[1]]-1);
            }
            
            for(int d=0; d<4; d++) {
                int nx = now[0] + dx[d];
                int ny = now[1] + dy[d];
                
                if(nx<=0 || nx>100 || ny<=0 || ny>100 || visited[nx][ny] != 0 || board[nx][ny] != 1) continue;
                visited[nx][ny] = visited[now[0]][now[1]] + 1;
                que.add(new int[]{nx, ny});
            }
        }
        
        return answer;
    }
}
```

## 코드1
```Java
import java.util.*;

class Solution {
    public int solution(int[][] rectangle, int characterX, int characterY, int itemX, int itemY) {
        int[][][] board = new int[51][51][4];
        int[][] visited = new int[51][51];
        
                                  //0  1  2  3  
        int[] dx = {-1, 1, 0, 0}; //↑, ↓, ←, →
        int[] dy = {0, 0, -1, 1};
        for(int[] rec : rectangle) {
            //위 ←, → (o), ↓ (x)
            //우 ↑, ↓ (o), ← (x)
            //하 ←, → (o), ↑ (x)
            //좌 ↑, ↓ (o), → (x)
            int x1 = rec[0], y1 = rec[1], x2 = rec[2], y2 = rec[3];
            
            for(int x=x1; x<=x2; x++) {
                for(int y=y1; y<=y2; y++) {
                    if(visited[x][y]!=0) continue;
                    if(x==x1 && y==y1) { //↓, →
                        if(board[x][y][1] != 2) {
                            board[x][y][1] = 1;
                        }
                        if(board[x][y][3] != 2) {
                            board[x][y][3] = 1;
                        }
                    } else if(x==x1 && y==y2) { //←, ↓
                        if(board[x][y][1] != 2) {
                            board[x][y][1] = 1;
                        }
                        if(board[x][y][2] != 2) {
                            board[x][y][2] = 1;
                        }
                    } else if(x==x2 && y==y2) { //↑, ←
                        if(board[x][y][0] != 2) {
                            board[x][y][0] = 1;
                        }
                        if(board[x][y][2] != 2) {
                            board[x][y][2] = 1;
                        }
                    } else if(x==x2 && y==y1) { // ↑, →
                        if(board[x][y][0] != 2) {
                            board[x][y][0] = 1;
                        }
                        if(board[x][y][3] != 2) {
                            board[x][y][3] = 1;
                        }
                    } else if(x==x1) {  //위 ←, → (o), ↓ (x)
                        if(board[x][y][2] != 2) {
                            board[x][y][2] = 1;
                        }
                        if(board[x][y][3] != 2) {
                            board[x][y][3] = 1;
                        }
                        board[x][y][1] = 2;
                    } else if(y==y2) { //우 ↑, ↓ (o), ← (x)
                        if(board[x][y][0] != 2) {
                            board[x][y][0] = 1;
                        }
                        if(board[x][y][1] != 2) {
                            board[x][y][1] = 1;
                        }
                        board[x][y][2] = 2;
                    } else if(x==x2) {  //하 ←, → (o), ↑ (x)
                        if(board[x][y][2] != 2) {
                            board[x][y][2] = 1;
                        }
                        if(board[x][y][3] != 2) {
                            board[x][y][3] = 1;
                        }
                        board[x][y][0] = 2;
                    } else if(y==y1) { //좌 ↑, ↓ (o), → (x) 
                        if(board[x][y][0] != 2) {
                            board[x][y][0] = 1;
                        }
                        if(board[x][y][1] != 2) {
                            board[x][y][1] = 1;
                        }
                        board[x][y][3] = 2;
                    } else {
                        visited[x][y] = -1;
                    }
                }
            }
        }
        
        int answer = Integer.MAX_VALUE;
        // bfs
        Queue<int[]> que = new ArrayDeque<>();
        que.add(new int[]{characterX, characterY});
        visited[characterX][characterY] = 1;
        
        while(!que.isEmpty()) {
            int[] now = que.poll();
            if(now[0]==itemX && now[1]==itemY) {
                answer = Math.min(answer, visited[now[0]][now[1]]-1);
                break;
            }
            
            for(int d=0; d<4; d++) {
                if(board[now[0]][now[1]][d] != 1) continue;
                
                int nx = now[0] + dx[d];
                int ny = now[1] + dy[d];
                
                if(nx<=0 || nx>50 || ny<=0 || ny>50 || visited[nx][ny] != 0) continue;
                visited[nx][ny] = visited[now[0]][now[1]] + 1;
                que.add(new int[]{nx, ny});
            }
        }
        
        return answer;
    }
}
```
