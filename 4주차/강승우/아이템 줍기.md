## 코드
```java
import java.util.*;

class Solution {
    public int solution(int[][] rectangle, int characterX, int characterY, int itemX, int itemY) {
        int answer = 0;
        
        int[] dx={-1,1,0,0};
        int[] dy={0,0,-1,1};
        
        // 두배를 해준 이유
        int start_x=characterX*2;
        int start_y=characterY*2;
        int end_x=itemX*2;
        int end_y=itemY*2;
        
        boolean[][] map= new boolean[103][103];
        
        for(int[] data:rectangle){
            //1.테두리 포함해서 직사각형 안까지 모두 true채우기
            for(int i=data[1]*2;i<=data[3]*2;i++){
              for(int j=data[0]*2;j<=data[2]*2;j++){
                    map[i][j]=true;
                  
              }  
            }
        }
        
        for(int[] data:rectangle){
            //2.테두리 제외하고 직사각형 내부 모두 false채우기
            for(int i=data[1]*2+1;i<data[3]*2;i++){
              for(int j=data[0]*2+1;j<data[2]*2;j++){
                    map[i][j]=false;
              }  
            }
        }
        
        
        Queue<Point> q= new LinkedList<>();
        int[][] visited= new int[103][103];
        q.add(new Point(start_x, start_y));
        visited[start_x][start_y]=0;
        while(!q.isEmpty()){
            Point p = q.poll();
            
            if (p.x==end_x && p.y==end_y){
                break;
            }
            // 네 뱡항으로 순회 하면서
            for(int i=0; i<4; i++){
                int nx=p.x+dx[i];
                int ny=p.y+dy[i];
                // 좌표계 범위를 벗어나지 않는 경우
                if (nx>0 && nx<103 && ny>0 && ny<103){
                    // 아직 방문하지 않았으면서, 경로가 있는 경우
                    if(visited[nx][ny]==0 && map[nx][ny]==true){
                        q.add(new Point(nx,ny));
                        visited[nx][ny]=visited[p.x][p.y]+1;
                    }
                }
            }
        }

        answer=visited[end_x][end_y]/2;
        
        
        return answer;
    }
    class Point{
        int x;
        int y;
        
        public Point(int x, int y){
            this.x=x;
            this.y=y;   
        }
    }
}
```
