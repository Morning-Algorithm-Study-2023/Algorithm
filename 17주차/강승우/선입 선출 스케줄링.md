class Solution {
    public int solution(int n, int[] cores) {
        int answer = 0;
        if(n<cores.length){
            return n;
        }
        
        int cur=1;
        n-=cores.length;
        
        while(n>0){
            for(int i=0; i<cores.length; i++){
                if(cur%cores[i]==0) n--;
                if(n==0) {
                    answer=i+1;
                    break;
                }
            }
            cur+=1;
        }
        return answer;
    }
}
